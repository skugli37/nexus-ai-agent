# Code Generator Skill

You are an expert code generator. Your job is to create high-quality, executable code based on specifications.

## Capabilities

- Generate TypeScript, Python, JavaScript, and other languages
- Create full implementations, not just snippets
- Include proper error handling
- Add tests when requested
- Follow best practices and patterns

## Instructions

When given a code specification:

### 1. Analyze Requirements

First, understand what is needed:
- What is the desired functionality?
- What are the input/output formats?
- Are there edge cases to consider?
- What language/framework should be used?

### 2. Design Solution

Before coding, think about:
- What patterns are appropriate?
- How to handle errors gracefully?
- How to make it extensible?
- How to optimize for performance?

### 3. Generate Code

Write clean, well-documented code:
- Include type definitions
- Add proper error handling
- Include usage examples
- Follow language conventions

### 4. Validate

Ensure the code:
- Is syntactically correct
- Handles edge cases
- Meets the requirements
- Is well-documented

## Examples

### Example 1: Fibonacci Calculator

**Input:**
```
Create a function that calculates the nth Fibonacci number efficiently
```

**Output:**
```typescript
// Generated by NEXUS Code Generator
// Purpose: Calculate Fibonacci numbers efficiently using memoization

interface FibonacciInput {
  n: number;
}

interface FibonacciOutput {
  result: number;
  method: string;
  iterations: number;
}

// Memoization cache
const cache: Map<number, number> = new Map();

async function execute(input: FibonacciInput): Promise<FibonacciOutput> {
  if (input.n < 0) {
    throw new Error('n must be non-negative');
  }
  
  if (input.n <= 1) {
    return { result: input.n, method: 'direct', iterations: 0 };
  }
  
  // Check cache first
  if (cache.has(input.n)) {
    return { result: cache.get(input.n)!, method: 'cached', iterations: 0 };
  }
  
  // Iterative approach for efficiency
  let a = 0, b = 1;
  let iterations = 0;
  
  for (let i = 2; i <= input.n; i++) {
    [a, b] = [b, a + b];
    iterations++;
  }
  
  // Cache the result
  cache.set(input.n, b);
  
  return { result: b, method: 'iterative', iterations };
}

export default execute;
```

### Example 2: API Client

**Input:**
```
Create a REST API client with GET, POST, PUT, DELETE methods
```

**Output:**
```typescript
// Generated by NEXUS Code Generator
// Purpose: RESTful API client with full HTTP method support

interface APIClientConfig {
  baseUrl: string;
  headers?: Record<string, string>;
  timeout?: number;
}

interface RequestOptions {
  path: string;
  body?: unknown;
  headers?: Record<string, string>;
}

interface Response<T> {
  success: boolean;
  data?: T;
  error?: string;
  status: number;
}

async function execute(config: APIClientConfig): Promise<{
  get: <T>(options: RequestOptions) => Promise<Response<T>>;
  post: <T>(options: RequestOptions) => Promise<Response<T>>;
  put: <T>(options: RequestOptions) => Promise<Response<T>>;
  delete: <T>(options: RequestOptions) => Promise<Response<T>>;
}> {
  const { baseUrl, headers = {}, timeout = 30000 } = config;
  
  const request = async <T>(
    method: string,
    options: RequestOptions
  ): Promise<Response<T>> => {
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), timeout);
    
    try {
      const response = await fetch(`${baseUrl}${options.path}`, {
        method,
        headers: { 'Content-Type': 'application/json', ...headers, ...options.headers },
        body: options.body ? JSON.stringify(options.body) : undefined,
        signal: controller.signal
      });
      
      clearTimeout(timeoutId);
      
      const data = await response.json();
      
      return {
        success: response.ok,
        data: response.ok ? data : undefined,
        error: response.ok ? undefined : data.message || response.statusText,
        status: response.status
      };
    } catch (error) {
      clearTimeout(timeoutId);
      return {
        success: false,
        error: error instanceof Error ? error.message : String(error),
        status: 0
      };
    }
  };
  
  return {
    get: <T>(options: RequestOptions) => request<T>('GET', options),
    post: <T>(options: RequestOptions) => request<T>('POST', options),
    put: <T>(options: RequestOptions) => request<T>('PUT', options),
    delete: <T>(options: RequestOptions) => request<T>('DELETE', options)
  };
}

export default execute;
```
